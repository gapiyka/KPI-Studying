package com.example.hapii

import android.graphics.Typeface
import android.os.Bundle
import android.view.View
import android.widget.Button
import android.widget.RadioGroup
import android.widget.TextView
import android.widget.Toast
import androidx.appcompat.app.AppCompatActivity

class MainActivity : AppCompatActivity() {

    override fun onCreate(savedInstanceState: Bundle?) {
        super.onCreate(savedInstanceState)
        setContentView(R.layout.activity_main)

        val textOutput = findViewById<TextView>(R.id.tOutput)
        val textInput = findViewById<TextView>(R.id.etInput)
        val buttonOk = findViewById<Button>(R.id.btnOk)
        val buttonCancel = findViewById<Button>(R.id.btnCancel)
        val radioGroup = findViewById<RadioGroup>(R.id.radioGroup)

        val fontsFaces = listOf(Typeface.SANS_SERIF, Typeface.MONOSPACE, Typeface.SERIF)

        buttonOk.setOnClickListener {
            val radioButtonID: Int = radioGroup.checkedRadioButtonId
            val radioButton: View = radioGroup.findViewById(radioButtonID)
            val idx: Int = radioGroup.indexOfChild(radioButton)
            val text = textInput.text.toString()

            if (text == "") {
                Toast.makeText(applicationContext, getString(R.string.empty_input),
                    Toast.LENGTH_SHORT,).show()
            }
            else {
                textOutput.text = text
                textOutput.typeface = fontsFaces[idx]
            }
        }

        buttonCancel.setOnClickListener {
            textOutput.text = getString(R.string.your_text_here)
        }
    }
}
